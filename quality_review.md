public void saveMPCode(SalesTable _salesTable, WMSPickingRoute _route, boolean _noSeparate = false)
{
    MPSalesPurchMarkCodeTable               codeTable;
    RabbitIntEngineImpInforSalesMPTable     salesMPTable;
    InventTable                             inventTable;
    RecordInsertList                        ril;
    int                                     kol = 0;
    ItemId                                  itemId = '';
    InventLocationId                        inventLocationId    = _salesTable.InventLocationId; //HNS_HNS_000002179_01, SMironov, 10.07.2025

    void writeRecord(RabbitIntEngineImpInforSalesMPTable _salesMPTable, inventTable _inventTable)
    {
        codeTable.clear();
        codeTable.initValue();
        switch (_salesMPTable.MPPackingType)
        {
            case InforMPPackingType::plt:
                codeTable.Agregation        = NoYes::Yes;
                codeTable.InforPalletteId   = NoYes::Yes;
                break;
            case InforMPPackingType::box,
                InforMPPackingType::pack:
                codeTable.Agregation        = NoYes::Yes;
                codeTable.InforPalletteId   = NoYes::No;
                break;
            default:
                codeTable.Agregation        = NoYes::No;
                codeTable.InforPalletteId   = NoYes::No;
                break;
        }
        codeTable.AgregationCount   = _salesMPTable.qty;
        codeTable.CustVendAC        = _salesTable.CustAccount;
        codeTable.ItemId            = _salesMPTable.ItemId;
        codeTable.MPCode            = _salesMPTable.mpCode;
        codeTable.pickingRouteID    = _route.pickingRouteID;
        codeTable.ProductName       = _inventTable.productName(CompanyInfo::languageId());
        codeTable.UnitId            = _inventTable.inventUnitId();
        codeTable.ProductGroupId    = _inventTable.MPGroupId;
        codeTable.Qty               = _salesMPTable.qty;
        codeTable.SalesPurch        = SalesPurch::Sales;
        codeTable.SalesPurchId      = _salesTable.SalesId;
        codeTable.InventLocationId  = inventLocationId; //HNS_HNS_000002179_01, SMironov, 10.07.2025
        //HNS_HNS_000002289_01, SMironov, 02.12.2025 ->
        codeTable.CorrectCode       = _salesMPTable.CorrectCode;
        codeTable.CaseSensitive     = _salesMPTable.CaseSensitive;
        //HNS_HNS_000002289_01, SMironov, 02.12.2025 <-
        ril.add(codeTable);
    }

    if (_route.RecId != 0)
    {
        ril = new RecordInsertList(tableNum(MPSalesPurchMarkCodeTable));
        while select * from salesMPTable
            order by itemId
            where salesMPTable.SalesId == _route.transRefId
                && ((_noSeparate == false
                        && salesMPTable.WMSPalletId == _route.EDCPalletCode)
                    ||
                    (_noSeparate == true))
        {
            if (itemId != salesMPTable.ItemId)
                inventTable = inventTable::find(salesMPTable.ItemId);
            if (salesMPTable.MPCode)
            {
                kol++;
                writeRecord(salesMPTable, inventTable);
            }
            itemId          = salesMPTable.ItemId;
        }
        ril.insertDatabase();
        info(strFmt("@MIK13877", _route.pickingRouteID, kol));
    }
}
